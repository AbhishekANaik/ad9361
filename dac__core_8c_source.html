145  dac_write(ADI_REG_RATECNTRL, ADI_RATE(3));
146 
147  dds_st.dac_clk = &ad9361_phy->clks[TX_SAMPL_CLK]->rate;
148 
149  dac_write(ADI_REG_CNTRL_1, 0);
150  switch (data_sel) {
151  case DATA_SEL_DDS:
152  dds_default_setup(DDS_CHAN_TX1_I_F1, 90000, 1000000, 4);
153  dds_default_setup(DDS_CHAN_TX1_I_F2, 90000, 1000000, 4);
154  dds_default_setup(DDS_CHAN_TX1_Q_F1, 0, 1000000, 4);
155  dds_default_setup(DDS_CHAN_TX1_Q_F2, 0, 1000000, 4);
156  dds_default_setup(DDS_CHAN_TX2_I_F1, 90000, 1000000, 4);
157  dds_default_setup(DDS_CHAN_TX2_I_F2, 90000, 1000000, 4);
158  dds_default_setup(DDS_CHAN_TX2_Q_F1, 0, 1000000, 4);
159  dds_default_setup(DDS_CHAN_TX2_Q_F2, 0, 1000000, 4);
160  dac_write(ADI_REG_CNTRL_2, ADI_DATA_SEL(DATA_SEL_DDS));
161  break;
162  case DATA_SEL_DMA:
163  tx_count = sizeof(sine_lut) / sizeof(uint16_t);
164  dac_write(ADI_REG_VDMA_FRMCNT, tx_count * 8);
165  for(index = 0; index < (tx_count * 2); index+=2)
166  {
167  index_i1 = index;
168  index_q1 = index + (tx_count / 4);
169  if(index_q1 >= (tx_count * 2))
170  index_q1 -= (tx_count * 2);
171  data_i1 = (sine_lut[index_i1 / 2] << 20);
172  data_q1 = (sine_lut[index_q1 / 2] << 4);
173  Xil_Out32(DAC_DDR_BASEADDR + index * 4, data_i1 | data_q1);
174 
175  index_i2 = index_i1 + (tx_count / 2);
176  index_q2 = index_q1 + (tx_count / 2);
177  if(index_i2 >= (tx_count * 2))
178  index_i2 -= (tx_count * 2);
179  if(index_q2 >= (tx_count * 2))
180  index_q2 -= (tx_count * 2);
181  data_i2 = (sine_lut[index_i2 / 2] << 20);
182  data_q2 = (sine_lut[index_q2 / 2] << 4);
183  Xil_Out32(DAC_DDR_BASEADDR + (index + 1) * 4, data_i2 | data_q2);
184  }
185  Xil_DCacheFlush();
186  vdma_write(XAXIVDMA_CR_OFFSET, 0x0);
187  vdma_write(XAXIVDMA_CR_OFFSET, XAXIVDMA_CR_TAIL_EN_MASK | XAXIVDMA_CR_RUNSTOP_MASK);
188  do {
189  vdma_read(XAXIVDMA_SR_OFFSET, &status);
190  }
191  while((status & 0x01) == 0x01);
192  vdma_write(XAXIVDMA_FRMSTORE_OFFSET, 0x01);
193  vdma_write(XAXIVDMA_MM2S_ADDR_OFFSET | XAXIVDMA_START_ADDR_OFFSET, DAC_DDR_BASEADDR);
194  vdma_write(XAXIVDMA_MM2S_ADDR_OFFSET | XAXIVDMA_STRD_FRMDLY_OFFSET, tx_count * 8);
195  vdma_write(XAXIVDMA_MM2S_ADDR_OFFSET | XAXIVDMA_HSIZE_OFFSET, tx_count * 8);
196  vdma_write(XAXIVDMA_MM2S_ADDR_OFFSET | XAXIVDMA_VSIZE_OFFSET, 1);
197  dac_write(ADI_REG_CNTRL_2, ADI_DATA_SEL(DATA_SEL_DMA));
198  break;
199  default:
200  break;
201  }
202  dac_write(ADI_REG_CNTRL_1, ADI_ENABLE);
203 }
